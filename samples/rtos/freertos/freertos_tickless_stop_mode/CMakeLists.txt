cmake_minimum_required(VERSION 3.13)

set(CONFIG_FREERTOS 1)
set(CONFIG_FREERTOS_TICKLESS 1)
set(CONFIG_FREERTOS_TICKLESS_USE_STOP_MODE 1)

find_package(hpm-sdk REQUIRED HINTS $ENV{HPM_SDK_BASE})
project(freertos_tickless_stop_mode)

sdk_inc(src)

sdk_app_src(src/freertos_tickless.c)

if("${HPM_SOC}" STREQUAL "HPM6750")

    sdk_compile_definitions(-DBOARD_CONSOLE_UART_CLK_NAME=clock_puart)
    sdk_compile_definitions(-DBOARD_CONSOLE_UART_IRQ=IRQn_PUART)
    sdk_app_src(src/hpm67xx/low_power_example.c)

    sdk_compile_definitions(-DBOARD_CONSOLE_UART_IRQ=IRQn_PUART)
    sdk_app_src(src/hpm63xx/low_power_example.c)
elseif("${HPM_SOC}" STREQUAL "HPM6280")
    sdk_compile_definitions(-DBOARD_CONSOLE_UART_BASE=HPM_PUART)
    sdk_compile_definitions(-DBOARD_CONSOLE_UART_CLK_NAME=clock_puart)

    sdk_app_src(src/hpm62xx/low_power_example.c)
elseif("${HPM_SOC}" STREQUAL "HPM6880")
    sdk_compile_definitions(-DBOARD_CONSOLE_UART_BASE=HPM_PUART)
    sdk_compile_definitions(-DBOARD_CONSOLE_UART_CLK_NAME=clock_puart)
