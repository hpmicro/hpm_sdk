if(APP_USE_ENET_PORT_COUNT EQUAL 1)
    set(_apply_config 0)
    if(DEFINED APP_USE_ENET_ITF_RMII AND APP_USE_ENET_ITF_RMII)
        set(_apply_config 1)
    elseif(DEFINED APP_USE_ENET_PHY_RTL8201 AND APP_USE_ENET_PHY_RTL8201)
        set(_apply_config 1)
    else()
        get_cmake_property(_vars VARIABLES)
        set(_itf_error 0)
        foreach(_v ${_vars})
            if(_v MATCHES "^APP_USE_ENET_ITF_" AND NOT _v STREQUAL "APP_USE_ENET_ITF_RMII")
                if(DEFINED ${_v} AND ${${_v}})
                    set(_itf_error 1)
                    break()
                endif()
            endif()
        endforeach()
        if(_itf_error)
            message(FATAL_ERROR "hpm6300evk does not support the configured ENET interface, only RMII is supported")
        else()
            set(_phy_error 0)
            foreach(_v ${_vars})
                if(_v MATCHES "^APP_USE_ENET_PHY_" AND NOT _v STREQUAL "APP_USE_ENET_PHY_RTL8201")
                    if(DEFINED ${_v} AND ${${_v}})
                        set(_phy_error 1)
                        break()
                    endif()
                endif()
            endforeach()
            if(_phy_error)
                message(FATAL_ERROR "hpm6300evk does not support the configured ENET PHY type, only RTL8201 is supported")
            else()
                message(WARNING "Neither APP_USE_ENET_ITF_RMII nor APP_USE_ENET_PHY_RTL8201 is set to a truthy value, using default RMII with RTL8201")
                set(_apply_config 1)
            endif()
        endif()
    endif()
    if(_apply_config)
        set(ENV{CONFIG_ENET_PHY_RTL8201} 1)
        sdk_compile_definitions(-DHPM_ENET_RMII=1)
    endif()
endif()
