set(ENV{CONFIG_PANEL_RGB_TM070RDH13} 1)
sdk_compile_definitions_ifdef(BUILD_FOR_SECONDARY_CORE BOARD_RUNNING_CORE=HPM_CORE1)

sdk_inc(.)
sdk_src(pinmux.c)
sdk_src(clock.c)
sdk_src(board.c)

if(APP_USE_ENET_PORT_COUNT EQUAL 2)
    set(ENV{CONFIG_ENET_PHY_RTL8211} 1)
    set(ENV{CONFIG_ENET_PHY_RTL8201} 1)
    sdk_compile_definitions(-DHPM_ENET_RGMII=1)
    sdk_compile_definitions(-DHPM_ENET_RMII=1)
elseif(APP_USE_ENET_PORT_COUNT EQUAL 1)
    set(_apply_rtl8211 0)
    set(_apply_rtl8201 0)
    if(DEFINED APP_USE_ENET_ITF_RGMII AND APP_USE_ENET_ITF_RGMII)
        set(_apply_rtl8211 1)
    elseif(DEFINED APP_USE_ENET_ITF_RMII AND APP_USE_ENET_ITF_RMII)
        set(_apply_rtl8201 1)
    elseif(DEFINED APP_USE_ENET_PHY_RTL8211 AND APP_USE_ENET_PHY_RTL8211)
        set(_apply_rtl8211 1)
    elseif(DEFINED APP_USE_ENET_PHY_RTL8201 AND APP_USE_ENET_PHY_RTL8201)
        set(_apply_rtl8201 1)
    else()
        get_cmake_property(_vars VARIABLES)
        set(_itf_error 0)
        foreach(_v ${_vars})
            if(_v MATCHES "^APP_USE_ENET_ITF_" AND NOT _v STREQUAL "APP_USE_ENET_ITF_RGMII" AND NOT _v STREQUAL "APP_USE_ENET_ITF_RMII")
                if(DEFINED ${_v} AND ${${_v}})
                    set(_itf_error 1)
                    break()
                endif()
            endif()
        endforeach()
        if(_itf_error)
            message(FATAL_ERROR "hpm6750evk2 does not support the configured ENET interface, only RGMII and RMII are supported")
        else()
            set(_phy_error 0)
            foreach(_v ${_vars})
                if(_v MATCHES "^APP_USE_ENET_PHY_" AND NOT _v STREQUAL "APP_USE_ENET_PHY_RTL8211" AND NOT _v STREQUAL "APP_USE_ENET_PHY_RTL8201")
                    if(DEFINED ${_v} AND ${${_v}})
                        set(_phy_error 1)
                        break()
                    endif()
                endif()
            endforeach()
            if(_phy_error)
                message(FATAL_ERROR "hpm6750evk2 does not support the configured ENET PHY type, only RTL8211 and RTL8201 are supported")
            else()
                message(WARNING "Neither RGMII/RMII nor RTL8211/RTL8201 is set to a truthy value, using default RGMII with RTL8211")
                set(_apply_rtl8211 1)
            endif()
        endif()
    endif()
    if(_apply_rtl8211)
        set(ENV{CONFIG_ENET_PHY_RTL8211} 1)
        sdk_compile_definitions(-DHPM_ENET_RGMII=1)
    elseif(_apply_rtl8201)
        set(ENV{CONFIG_ENET_PHY_RTL8201} 1)
        sdk_compile_definitions(-DHPM_ENET_RMII=1)
    endif()
endif()
