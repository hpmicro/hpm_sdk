sdk_inc(.)
sdk_src(pinmux.c)
sdk_src(board.c)
sdk_src(clock.c)

set(ENV{CONFIG_PANEL_RGB_TM070RDH13} 1)
set(ENV{CONFIG_PANEL_LVDS_CC10128007} 0)
set(ENV{CONFIG_PANEL_MIPI_MC10128007_31B} 0)
set(ENV{CONFIG_PANEL_LVDS_TM103XDGP01} 0)

if(APP_USE_ENET_PORT_COUNT EQUAL 1)
    set(_apply_config 0)
    if(DEFINED APP_USE_ENET_ITF_RGMII AND APP_USE_ENET_ITF_RGMII)
        set(_apply_config 1)
    elseif(DEFINED APP_USE_ENET_PHY_RTL8211 AND APP_USE_ENET_PHY_RTL8211)
        set(_apply_config 1)
    else()
        get_cmake_property(_vars VARIABLES)
        set(_itf_error 0)
        foreach(_v ${_vars})
            if(_v MATCHES "^APP_USE_ENET_ITF_" AND NOT _v STREQUAL "APP_USE_ENET_ITF_RGMII")
                if(DEFINED ${_v} AND ${${_v}})
                    set(_itf_error 1)
                    break()
                endif()
            endif()
        endforeach()
        if(_itf_error)
            message(FATAL_ERROR "hpm6800evk does not support the configured ENET interface, only RGMII is supported")
        else()
            set(_phy_error 0)
            foreach(_v ${_vars})
                if(_v MATCHES "^APP_USE_ENET_PHY_" AND NOT _v STREQUAL "APP_USE_ENET_PHY_RTL8211")
                    if(DEFINED ${_v} AND ${${_v}})
                        set(_phy_error 1)
                        break()
                    endif()
                endif()
            endforeach()
            if(_phy_error)
                message(FATAL_ERROR "hpm6800evk does not support the configured ENET PHY type, only RTL8211 is supported")
            else()
                message(WARNING "Neither APP_USE_ENET_ITF_RGMII nor APP_USE_ENET_PHY_RTL8211 is set to a truthy value, using default RGMII with RTL8211")
                set(_apply_config 1)
            endif()
        endif()
    endif()
    if(_apply_config)
        set(ENV{CONFIG_ENET_PHY_RTL8211} 1)
        sdk_compile_definitions(-DHPM_ENET_RGMII=1)
    endif()
endif()
